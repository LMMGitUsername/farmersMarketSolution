$(document).ready(function() {
  //generate a quote as soon as the page loads
  $.get("https://api.whatdoestrumpthink.com/api/v1/quotes/random", function(data, status){
  $(".well").html(data.message);
    
  var mydiv = document.getElementById("tweet");
  var aTag = document.createElement('a');
  aTag.className = "tweetText";
  var tweetHref = "https://twitter.com/intent/tweet?text=" + data.message;
  aTag.setAttribute('href',tweetHref);
  aTag.innerHTML = tweetHref;
  mydiv.appendChild(aTag);  
});
  //generate a quote when the user clicks the Generate... button
  $("#generateNewQuote").on("click", function(){
    $.get("https://api.whatdoestrumpthink.com/api/v1/quotes/random", function(data, status){
    $(".well").html(data.message);
    var tweetHref = "https://twitter.com/intent/tweet?text=" + data.message;  
    $(".tweetText").attr('href',tweetHref);
});
});
  

  
});
